    function loadData() {
      const savedUser = localStorage.getItem('khaffef_user');
      if(savedUser) { try { userData = JSON.parse(savedUser); } catch(e){} }
      if (!userData.activityLevel) userData.activityLevel = '1.375';

    function calculateBMR() {
      const { age, weight, height, gender } = userData;
      return gender === 'male' ? (10 * weight) + (6.25 * height) - (5 * age) + 5 : (10 * weight) + (6.25 * height) - (5 * age) - 161;
    }
    function calculateTDEE() {
      return Math.round(calculateBMR() * (parseFloat(userData.activityLevel) || 1.375));
    }
    function getMETFromSpeed(speedKmh) {
      if (speedKmh < 3.5) return 2.5; if (speedKmh < 5.6) return 3.5;
      if (speedKmh < 7.6) return 5.0; if (speedKmh < 9.6) return 7.0; return 9.5;
    }
    function calculateWalkingCalories(distanceMeters, minutes, seconds) {
      const weight = userData.weight, totalMinutes = minutes + seconds / 60;
      if (totalMinutes <= 0) return { calories: 0, speed: 0, met: 2.5 };
      const distanceKm = distanceMeters / 1000, timeHours = totalMinutes / 60;
      const speedKmh = distanceKm / timeHours, met = getMETFromSpeed(speedKmh);
      return { calories: Math.round((met * 3.5 * weight) / 200 * totalMinutes), speed: speedKmh, met };
    }

    async function analyzeFoodImage(base64Image) {
      try {
        const prompt = `أنت خبير تغذية. حلل صورة الطعام هذه وقم فقط بتقدير عدد السعرات الحرارية الإجمالية في الوجبة.
أعد الرد بتنسيق JSON صارم يحتوي على الحقول التالية:
{
  "meal_description": "وصف مختصر للوجبة بالعربية",
  "estimated_calories": رقم صحيح
}
لا تقم بتضمين أي نص آخر خارج JSON.`;
        const response = await fetch(GEMINI_URL, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            contents: [{ parts: [{ text: prompt }, { inline_data: { mime_type: 'image/jpeg', data: base64Image } }] }]
          })
        });
        const data = await response.json();
        if (!data.candidates || !data.candidates[0]?.content?.parts?.[0]?.text) {
          throw new Error('لم يتم استلام رد صحيح من Gemini');
        }
        let jsonText = data.candidates[0].content.parts[0].text;
        jsonText = jsonText.replace(/```json|```/g, '').trim();
        const result = JSON.parse(jsonText);
        const calories = Math.round(result.estimated_calories) || 0;
        const desc = result.meal_description || 'وجبة';
        analysisStatus.textContent = '';
        analysisResultDiv.classList.remove('hidden');
        analysisResultDiv.innerHTML = `
          <p><strong>🍽️ ${desc}</strong></p>
          <p style="font-size:28px; margin:8px 0;">🔥 ${calories} سعرة</p>
          <div style="display:flex; gap:10px; margin-top:12px;">
            <button class="btn-primary" id="addAnalyzedCaloriesBtn" style="flex:1;">✅ إضافة</button>
            <button class="btn-secondary" id="cancelAnalyzedCaloriesBtn" style="flex:1;">❌ إلغاء</button>
          </div>
        `;
        document.getElementById('addAnalyzedCaloriesBtn').addEventListener('click', () => {
          const today = getTodayStr();
          if (!dailyLogs[today]) dailyLogs[today] = { eaten:0, burned:0 };
          dailyLogs[today].eaten += calories;
          saveLogs(); refreshUI();
          closeCamera();
        });
        document.getElementById('cancelAnalyzedCaloriesBtn').addEventListener('click', closeCamera);
      } catch (error) {
        analysisStatus.textContent = '❌ حدث خطأ: ' + error.message;
        console.error(error);
      }
    }

    function showDeficitDetails() {
      const today = getTodayStr(), log = dailyLogs[today]||{eaten:0,burned:0};
      const bmr = calculateBMR(), tdee = calculateTDEE(), totalOut = tdee + log.burned, deficit = totalOut - log.eaten;
      const totalDeficit = calculateTotalDeficit();
      const actLabels = {'1.2':'قليل','1.375':'خفيف','1.55':'متوسط','1.725':'عالٍ','1.9':'مكثف'};
      document.getElementById('deficitDetailsText').innerHTML = `
        <p>🔥 BMR: ${Math.round(bmr)} سعرة</p><p>⚡ النشاط: ${actLabels[userData.activityLevel]} (${userData.activityLevel})</p>
        <p>📊 TDEE: ${tdee} سعرة</p><p>🚶 مشي: ${log.burned}</p><p>🍽️ أكل: ${log.eaten}</p>
        <p><strong>${deficit>=0?'العجز':'الفائض'}: ${Math.abs(Math.round(deficit))} سعرة</strong></p>
        <hr><p>📅 إجمالي العجز التراكمي: ${Math.round(totalDeficit)} سعرة</p><p>⚖️ خسارة متوقعة: ${(totalDeficit/7700).toFixed(2)} كجم</p>`;
    }

    function renderCalendar() {
      const today = new Date(), y = today.getFullYear(), m = today.getMonth();
      const firstDay = new Date(y,m,1), lastDay = new Date(y,m+1,0), start = firstDay.getDay();
      let html = `<div class="calendar-month">${new Intl.DateTimeFormat('ar-EG',{month:'long',year:'numeric'}).format(today)}</div><div class="calendar-grid">`;
      ['سبت','أحد','إثنين','ثلاثاء','أربعاء','خميس','جمعة'].forEach(d=>html+=`<div style="font-weight:bold;">${d}</div>`);
      let day=1;
      for(let i=0;i<6;i++) for(let j=0;j<7;j++) {
        if(i===0 && j<start) html+='<div></div>';
        else if(day<=lastDay.getDate()) {
          const ds = `${y}-${String(m+1).padStart(2,'0')}-${String(day).padStart(2,'0')}`;
          const log = dailyLogs[ds], deficit = log ? computeDeficit(log.eaten, log.burned) : null;
          let cls = 'cal-day'; if(deficit!==null) cls+= deficit>=0?' deficit':' surplus';
          if(ds===getTodayStr()) cls+=' today-ring';
          html+=`<div class="${cls}" data-date="${ds}">${day++}</div>`;
        } else html+='<div></div>';
      }
      html+='</div>'; document.getElementById('calendarContent').innerHTML = html;
      document.querySelectorAll('.cal-day[data-date]').forEach(el=>el.addEventListener('click',function(){
        const ds=this.dataset.date, log=dailyLogs[ds];
        if(log){ const def=computeDeficit(log.eaten,log.burned); alert(`${ds}\n🍽️ ${log.eaten}\n🔥 ${log.burned}\n${def>=0?'عجز':'فائض'}: ${Math.abs(Math.round(def))}`);}
        else alert('لا بيانات');
      }));
    }
